# C-1. Threat Modeling Spec

`04. Threat Modeling`의 기술 세부 문서입니다. Step A~D의 실행 가능한 규칙 엔진, 실제 소스코드, 손계산 검증을 담습니다. 개념·근거·쉬운 설명은 `04. Threat Modeling`을 참고하고, 이 문서는 "정확히 어떻게 계산되는가"에 집중합니다.

입력은 `03. Sensor Abstraction Layer`가 만드는 공통 envelope(`schema_version`/`id`/`ts`/`channels[]`, 채널별 `channel`/`state`/`quality`/`quality_delta`/`payload`)입니다. 전체 스펙은 `A-1. 추상 결과 세부 내용` 참고.

## 1. 선언적 룰 엔진 — Step B

기존 방식은 위협마다 if문을 코드에 하드코딩했습니다. 위협이 늘어날 때마다 함수를 계속 고쳐야 해서, 아래처럼 **테이블 자체가 조건 함수를 갖는 선언적 구조**로 바꿨습니다. 새 위협을 추가할 때는 `SIGNAL_TO_THREAT_RULES`에 행만 추가하면 되고, `step_b_signal_matching` 함수는 손대지 않습니다.

```python
def _t3_proximity(s):
    return s["state"] == "anomaly" and s["payload"].get("weapon_shape") is True

def _t3_acoustic(s):
    return s["payload"].get("event_type") == "gunshot"

def _t1_position(s):
    return s["payload"].get("gps_imu_residual_m", 0) > 5.0

def _t1_rf(s):
    return s["payload"].get("wideband_anomaly") is True

def _t2_link_integrity(s):
    return (s["payload"].get("checksum_fail_rate", 0) > CHECKSUM_FAIL_RATE_THRESHOLD
            or s["payload"].get("seq_gap_count", 0) > 0)

def _t2_encryption(s):
    return s["payload"].get("downgrade_detected") is True

def _t7_obstacle(s):
    # 고정거리(10.0m) 대신 충돌예상시간(time-to-collision) 기준 — 기체 속도와 무관하게
    # 물리적으로 일관된 판정이 되도록 재설계됨(알려진 한계 #1, 이번 라운드 해결)
    distance_m = s["payload"].get("distance_m")
    closure_rate_mps = s["payload"].get("closure_rate_mps", 0)
    # 음수 거리는 물리적으로 무의미(음수 TTC → 오탐) — 배제.
    if distance_m is None or distance_m < 0 or closure_rate_mps <= 0:
        return False
    time_to_collision_s = distance_m / closure_rate_mps
    return time_to_collision_s < TIME_TO_COLLISION_THRESHOLD_S

def _t4_proximity(s):
    return s["payload"].get("class") in ("person", "vehicle") and s["payload"].get("closing") is True

def _t4_phase_mismatch(s):
    return s["payload"].get("match") is False

def _t4_link_abnormal(s):
    return s["state"] != "normal"

def _t5_quality_delta(s):
    return s.get("quality_delta", 0) < -0.3

SIGNAL_TO_THREAT_RULES = [
    {"threat": "T3", "conditions": [("proximity_object", _t3_proximity)]},
    {"threat": "T3", "conditions": [("acoustic_event", _t3_acoustic)]},
    {"threat": "T1", "conditions": [("position_consistency", _t1_position)]},
    {"threat": "T1", "conditions": [("rf_spectrum", _t1_rf)]},
    {"threat": "T2", "conditions": [("link_integrity", _t2_link_integrity)]},
    {"threat": "T2", "conditions": [("encryption_status", _t2_encryption)]},
    {"threat": "T7", "conditions": [("obstacle_proximity", _t7_obstacle)]},
    {"threat": "T4", "conditions": [                      # 다중채널 AND
        ("proximity_object", _t4_proximity),
        ("mission_phase", _t4_phase_mismatch),
        ("link_status", _t4_link_abnormal),
    ]},
    {"threat": "T5", "conditions": [("proximity_object", _t5_quality_delta)]},
    {"threat": "T5", "conditions": [("terrain_class", _t5_quality_delta)]},
]
```

각 rule의 `conditions`가 전부 참이어야 매칭됩니다. 길이 1이면 단일채널 조건(T1/T2/T3/T7/T5), 길이 3이면 T4처럼 여러 채널이 동시에 참이어야 하는 AND 조합입니다 — **같은 매칭 엔진 코드로 둘 다 처리**되고, Step C는 매칭 결과가 단일 규칙에서 왔는지 다중채널 규칙에서 왔는지 구분하지 않습니다.

```python
def step_b_signal_matching(sensor_snapshot):
    by_channel = {s["channel"]: s for s in sensor_snapshot}
    matched = []
    for rule in SIGNAL_TO_THREAT_RULES:
        conds = rule["conditions"]
        if not all(ch in by_channel for ch, _ in conds):
            continue  # 이번 사이클에 필요한 채널이 안 들어온 경우 평가 불가
        if all(cond(by_channel[ch]) for ch, cond in conds):
            for ch, _ in conds:
                matched.append((ch, rule["threat"], by_channel[ch]["quality"]))
    return matched
```

T4 규칙이 매칭되면 `matched`에 `(proximity_object, T4, q)`, `(mission_phase, T4, q)`, `(link_status, T4, q)` 세 항목이 한꺼번에 들어갑니다 — 이후 Step C는 이걸 T3의 2채널 매칭과 똑같은 방식으로 처리합니다.

## 2. 채널 가중치 — 하이브리드 구조

기본은 채널 전역값이고, 필요하면 `(channel, threat)` 쌍으로 오버라이드할 수 있습니다(지금은 오버라이드가 하나도 없어 전부 기본값을 씁니다).

```python
CHANNEL_WEIGHTS = {
    "proximity_object": 0.40,
    "position_consistency": 0.35,
    "link_integrity": 0.35,
    "obstacle_proximity": 0.35,
    "encryption_status": 0.35,
    "acoustic_event": 0.30,
    "rf_spectrum": 0.25,
    "terrain_class": 0.25,
    "mission_phase": 0.25,
    "link_status": 0.15,
}
CHANNEL_WEIGHT_DEFAULT = 0.20

def channel_weight(channel, threat=None):
    if threat is not None and (channel, threat) in CHANNEL_WEIGHTS:
        return CHANNEL_WEIGHTS[(channel, threat)]
    return CHANNEL_WEIGHTS.get(channel, CHANNEL_WEIGHT_DEFAULT)
```

이 값들은 팀이 정한 설계값입니다. RAG 코퍼스가 쌓이면 로지스틱 회귀로 재학습할 수 있는데 — 그 방법과 mock 데이터 검증 결과는 6절에 정리했습니다.

## 3. Step C — quality 2단 제외, 결정론적/AI confidence, 국면 배수, 후보 선정

```python
import math

def _logit(q):
    q = min(max(q, 0.001), 0.999)   # 0/1 극단값에서 math domain error 방지
    return math.log(q / (1 - q))

def _sigmoid(x):
    return 1 / (1 + math.exp(-x))

def step_c_kill_chain(matched_signals, declared_phase=None):
    if not matched_signals:
        return {"candidates": [], "primary": None}

    threats = {}
    for channel, threat, quality in matched_signals:
        threats.setdefault(threat, {})[channel] = quality   # 채널 중복 제거

    candidates = []
    for threat_event, ch_quality in threats.items():
        # ── quality 2단 제외 ──
        effective = []
        for ch, q in ch_quality.items():
            w = channel_weight(ch, threat_event)
            if w < W_MIN:          # 구조적 하한 — 항상 제외
                continue
            if q < Q_MIN:           # 열화 하한 — 이번만 제외
                continue
            effective.append((ch, w, q))

        if not effective:
            continue                # 이 위협은 이번 사이클 candidates에서 탈락

        # ── 결정론적 confidence ──
        match_count = len(effective)
        det_confidence = CONFIDENCE_BY_MATCH_COUNT.get(
            match_count, CONFIDENCE_BY_MATCH_COUNT[max(CONFIDENCE_BY_MATCH_COUNT)])
        avg_weight = sum(w for _, w, _ in effective) / match_count
        kill_chain_stage = ("후기" if avg_weight >= 0.35 and match_count >= 2
                             else "중기" if match_count >= 1 else "초기")

        # ── AI 강화판: 로그오즈 결합 ──
        log_odds_total = sum(w * _logit(q) for _, w, q in effective)
        ai_confidence = _sigmoid(log_odds_total)

        # ── 교차검증 ──
        if abs(ai_confidence - det_confidence) <= CROSS_CHECK_TOLERANCE:
            confidence, confidence_source = ai_confidence, "ai"
        else:
            confidence, confidence_source = det_confidence, "deterministic"

        # ── 국면 배수 ──
        multiplier = PHASE_THREAT_MULTIPLIER.get((declared_phase, threat_event), 1.0)
        confidence = min(confidence * multiplier, CONFIDENCE_CAP)

        candidates.append({
            "threat_event": threat_event,
            "match_count": match_count,
            "confidence": round(confidence, 3),
            "confidence_source": confidence_source,
            "kill_chain_stage": kill_chain_stage,
            "potential_outcome": step_d_outcome(threat_event),
            "_avg_weight": avg_weight,   # primary 선정용 내부 필드
        })

    # ── 후보 전체 중 대표(primary) 선정 ──
    primary = None
    if candidates:
        primary = max(candidates, key=lambda c: (c["match_count"], c["_avg_weight"]))
        primary = {k: v for k, v in primary.items() if k != "_avg_weight"}
    for c in candidates:
        c.pop("_avg_weight", None)

    return {"candidates": candidates, "primary": primary}
```

상수:

```python
CONFIDENCE_BY_MATCH_COUNT = {1: 0.7, 2: 0.9, 3: 0.95}
W_MIN = 0.20
Q_MIN = 0.65
CROSS_CHECK_TOLERANCE = 0.15
CONFIDENCE_CAP = 0.95
PHASE_THREAT_MULTIPLIER = {
    ("LOITER_ROI", "T3"): 1.1,   # 정찰중 정지상태 → 근접소화기 위협 상향
    ("LOITER_ROI", "T7"): 0.9,   # 정찰중 안전고도 유지 → 지형충돌 위협 하향
    ("LOITER_ROI", "T4"): 1.1,   # 정지상태 → 포획 시도에 취약
    ("RTL", "T3"): 0.9,          # 이탈 중 → 근접위협 하향
    ("LAND", "T7"): 1.2,         # 착륙접근 저고도·지형근접 → 지형충돌 가장 상향
    ("LAND", "T4"): 1.2,         # 저속·저고도 → 포획 시도에 가장 취약
    ("TAKEOFF", "T7"): 1.1,      # 이륙 직후 저고도 → 지형충돌 소폭 상향
    ("TAKEOFF", "T4"): 1.1,      # 저속 구간 → 포획 시도에 취약
```

## 6. 채널 가중치 재학습 — retrain_weights.py

Step C의 로그오즈 결합식(`log_odds = Σ weight(c) × logit(quality(c))`)은 형태상 로지스틱 회귀와 동일합니다. 각 채널의 `logit(quality)`를 피처로, 실제 위협 확인 여부(RAG 코퍼스의 사후 라벨)를 타깃으로 보면, `weight(c)`를 경사하강법으로 재적합할 수 있습니다. 외부 ML 라이브러리 없이 순수 Python(math만 사용)으로 구현해 온보드/오프보드 어디서든 의존성 부담이 없습니다.

```python
import math
import random

def _logit(q):
    q = min(max(q, 1e-6), 1 - 1e-6)
    return math.log(q / (1 - q))

def _sigmoid(x):
    return 1 / (1 + math.exp(-x))

def retrain_channel_weights(labeled_examples, initial_weights, lr=0.05, epochs=500, l2=0.01):
    """
    labeled_examples: [{"channels": [(channel, quality), ...], "confirmed": bool}, ...]
        RAG 코퍼스에서 뽑은 (이번 사이클 매칭 채널들의 quality, 실제 위협 확인 여부) 라벨 데이터
    initial_weights: dict channel -> weight (현재 CHANNEL_WEIGHTS, cold-start 시작값)
    반환: 재학습된 dict channel -> weight (사람 검토 전 초안 — 자동 반영 안 함)
    """
    channels = sorted(initial_weights.keys())
    w = {c: initial_weights[c] for c in channels}

    for _ in range(epochs):
        grad = {c: 0.0 for c in channels}
        for ex in labeled_examples:
            x = {c: 0.0 for c in channels}
            for ch, q in ex["channels"]:
                if ch in x:
                    x[ch] = _logit(q)
            z = sum(w[c] * x[c] for c in channels)
            pred = _sigmoid(z)
            label = 1.0 if ex["confirmed"] else 0.0
            err = pred - label
            for c in channels:
                grad[c] += err * x[c]
        n = len(labeled_examples)
        for c in channels:
            g = grad[c] / n + l2 * w[c]      # L2 정규화 — 표본 적을 때 과적합 방지
            w[c] -= lr * g
            w[c] = max(0.0, w[c])            # 가중치는 항상 0 이상(음의 신뢰도 없음)
    return w
```

운영 방식: 임무 20건 누적마다 배치로 재학습, 사람이 검토·승인한 뒤 반영(자동 반영 안 함). 그 전까지는 CHANNEL_WEIGHTS의 손으로 정한 값을 그대로 씁니다.

**의도한 효과**: `rf_spectrum`은 설계값(0.25)보다 실제로 신뢰도가 낮다고 시뮬레이션했고, 재학습 결과 0.081로 정확히 하향 조정됐습니다 — 재학습 메커니즘 자체는 의도대로 작동합니다.

**캐비어트(정직하게 기록)**: `acoustic_event`가 함께 0.30→0.046으로 급락했는데, 이건 의도한 시뮬레이션이 아닙니다. T3 패턴 표본이 8건뿐이고 `proximity_object`와 `acoustic_event`가 매 사례마다 항상 함께 등장(상관관계 100%)하다 보니, 경사하강법이 두 채널 중 하나(`proximity_object`)에 가중치를 몰아주고 나머지(`acoustic_event`)를 거의 0으로 깎아도 훈련 손실은 똑같이 줄어듭니다 — 상관된 피처가 적은 표본으로 학습될 때 흔히 나타나는 로지스틱 회귀의 불안정성입니다. 실제 운영 시에는 (a) 최소 표본 수 하한을 두거나, (b) 채널 쌍의 상관관계를 사전 점검하거나, (c) 정규화 강도를 채널 그룹별로 다르게 주는 보완이 필요합니다 — 지금은 재학습 결과를 사람이 검토 후 승인하는 절차가 이미 있어 이런 결과가 그대로 반영되지는 않지만, 향후 표본이 늘면 재검토할 사안으로 남겨둡니다.

### threat_catalog_v2.py

```python
# -*- coding: utf-8 -*-
"""D4D 위협모델링(04) 결정론적 룩업테이블 v2"""

THREAT_CATALOG = {
    "T1": "EW/GPS스푸핑", "T2": "사이버/C2하이재킹", "T3": "근접소화기",
    "T4": "물리포획", "T5": "레이저", "T6": "환경노출도(배경)", "T7": "지형충돌/CFIT",
```

```python
# -*- coding: utf-8 -*-
"""위협모델링(04) Step A~D v2"""
import math
from threat_catalog_v2 import (
    SIGNAL_TO_THREAT_RULES, CHANNEL_WEIGHTS, CHANNEL_WEIGHT_DEFAULT, channel_weight,
    CONFIDENCE_BY_MATCH_COUNT, W_MIN, Q_MIN, CROSS_CHECK_TOLERANCE,
    PHASE_THREAT_MULTIPLIER, CONFIDENCE_CAP, POTENTIAL_OUTCOME,
)


def _logit(q):
    q = min(max(q, 0.001), 0.999)
    return math.log(q / (1 - q))


def _sigmoid(x):
    return 1 / (1 + math.exp(-x))


def step_a_mission_phase(sensor_snapshot):
    mp = next(s for s in sensor_snapshot if s["channel"] == "mission_phase")
    return mp["payload"]["declared"], mp["payload"]["mission_phase_confidence"]


def step_b_signal_matching(sensor_snapshot):
    by_channel = {s["channel"]: s for s in sensor_snapshot}
    matched = []
    for rule in SIGNAL_TO_THREAT_RULES:
        conds = rule["conditions"]
        if not all(ch in by_channel for ch, _ in conds):
            continue
        if all(cond(by_channel[ch]) for ch, cond in conds):
            for ch, _ in conds:
                matched.append((ch, rule["threat"], by_channel[ch]["quality"]))
    return matched


def step_b_background_exposure(sensor_snapshot):
    terrain = next((s for s in sensor_snapshot if s["channel"] == "terrain_class"), None)
    if terrain is None:
        return None
    return terrain["payload"].get("exposure_score")


def step_c_kill_chain(matched_signals, declared_phase=None):
    if not matched_signals:
        return {"candidates": [], "primary": None}

    threats = {}
    for channel, threat, quality in matched_signals:
        threats.setdefault(threat, {})[channel] = quality

    candidates = []
    for threat_event, ch_quality in threats.items():
        effective = []
        for ch, q in ch_quality.items():
            w = channel_weight(ch, threat_event)
            if w < W_MIN:
                continue
            if q < Q_MIN:
                continue
            effective.append((ch, w, q))

        if not effective:
            continue

        match_count = len(effective)
        det_confidence = CONFIDENCE_BY_MATCH_COUNT.get(
            match_count, CONFIDENCE_BY_MATCH_COUNT[max(CONFIDENCE_BY_MATCH_COUNT)])
        avg_weight = sum(w for _, w, _ in effective) / match_count
        kill_chain_stage = ("후기" if avg_weight >= 0.35 and match_count >= 2
                             else "중기" if match_count >= 1 else "초기")

        log_odds_total = sum(w * _logit(q) for _, w, q in effective)
        ai_confidence = _sigmoid(log_odds_total)

        if abs(ai_confidence - det_confidence) <= CROSS_CHECK_TOLERANCE:
            confidence, confidence_source = ai_confidence, "ai"
        else:
            confidence, confidence_source = det_confidence, "deterministic"

        multiplier = PHASE_THREAT_MULTIPLIER.get((declared_phase, threat_event), 1.0)
        confidence = min(confidence * multiplier, CONFIDENCE_CAP)

        candidates.append({
            "threat_event": threat_event,
            "match_count": match_count,
            "confidence": round(confidence, 3),
            "confidence_source": confidence_source,
            "kill_chain_stage": kill_chain_stage,
            "potential_outcome": step_d_outcome(threat_event),
            "_avg_weight": avg_weight,
        })

    primary = None
    if candidates:
        primary = max(candidates, key=lambda c: (c["match_count"], c["_avg_weight"]))
        primary = {k: v for k, v in primary.items() if k != "_avg_weight"}
    for c in candidates:
        c.pop("_avg_weight", None)

    return {"candidates": candidates, "primary": primary}


def step_d_outcome(threat_event):
    return POTENTIAL_OUTCOME.get(threat_event, "unknown")


def run_threat_modeling(sensor_snapshot, mission_brief=None):
    declared_phase, phase_conf = step_a_mission_phase(sensor_snapshot)
    matched = step_b_signal_matching(sensor_snapshot)
    result = step_c_kill_chain(matched, declared_phase=declared_phase)
    background_exposure_score = step_b_background_exposure(sensor_snapshot)
    return {
        "declared_phase": declared_phase,
        "mission_phase_confidence": phase_conf,
        "candidates": result["candidates"],
        "primary": result["primary"],
        "background_exposure_score": background_exposure_score,
    }
```
