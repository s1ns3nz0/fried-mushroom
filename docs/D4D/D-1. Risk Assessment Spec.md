# D-1. Risk Assessment Spec

`05. Risk Assessment`의 기술 세부 문서입니다. 실행 가능한 코드, 손계산 검증, 파라미터 출처를 담습니다. 개념·근거·쉬운 설명은 `05. Risk Assessment`를 참고하고, 이 문서는 "정확히 어떻게 계산되는가"에 집중합니다.

입력은 `04. Threat Modeling`의 `run_threat_modeling()` 출력(`candidates[]`, `background_exposure_score`)과 `mission_brief`(`mission_context`, `posture`, `spare_asset_available`, `battery_pct`, `link_quality`)입니다.

## 2. BASE_RATE — threat_category별 컨텍스트 민감도

```python
MISSION_CONTEXTS = ["정찰", "타격", "호송", "수송"]

BASE_RATE_CONTEXT_SENSITIVE = {
    ("T3", "정찰"): 0.15, ("T3", "타격"): 0.35, ("T3", "호송"): 0.20, ("T3", "수송"): 0.10,
    ("T4", "정찰"): 0.08, ("T4", "타격"): 0.20, ("T4", "호송"): 0.12, ("T4", "수송"): 0.05,
}
BASE_RATE_CONTEXT_INVARIANT = {
    "T1": 0.12, "T2": 0.10, "T5": 0.08, "T7": 0.10,
}

def get_base_rate(threat_event, mission_context):
    if THREAT_CATEGORY[threat_event] == "PHYSICAL":
        return BASE_RATE_CONTEXT_SENSITIVE[(threat_event, mission_context)]
    return BASE_RATE_CONTEXT_INVARIANT[threat_event]
```

PHYSICAL(T3/T4)만 실제로 컨텍스트별로 다른 값을 갖고(8칸), REMOTE/NAVIGATION(T1/T2/T5/T7)은 컨텍스트 무관 단일값(4칸)이라 논리적으로는 7×4=28칸이지만 실제 채워야 하는 숫자는 12개뿐입니다.

## 3. L(발생가능성) — 등급화 + posture

```python
def l_value_to_class(l_value: float) -> str:
    if l_value >= 0.5: return "A"
    if l_value >= 0.3: return "B"
    if l_value >= 0.15: return "C"
    if l_value >= 0.05: return "D"
    if l_value > 0: return "E"
    return "F"

CLASS_ORDER = ["A", "B", "C", "D", "E", "F"]

def shift_class(cls: str, steps: int) -> str:
    idx = max(0, CLASS_ORDER.index(cls) - steps)
    return CLASS_ORDER[idx]

def posture_shift_steps(posture: dict, threat: str) -> int:
    kinetic_ew = {"T1", "T3", "T4", "T5", "T7"}
    cyber = {"T2"}
    level = min(posture.get("watchcon", 5), posture.get("defcon", 5)) if threat in kinetic_ew else posture.get("infocon", 5)
    if level >= 4: return 0
    if level == 3: return 1
    return 2
```

## 4. S(심각도) — override

```python
OUTCOME_TO_SEVERITY = {
    "attrition_kill": "Critical", "mission_abort": "Marginal", "hull_loss": "Catastrophic",
}
SEVERITY_ORDER = {"Catastrophic": 1, "Critical": 2, "Marginal": 3, "Negligible": 4}
SEVERITY_LABEL_BY_NUM = {v: k for k, v in SEVERITY_ORDER.items()}

def apply_severity_override(severity_num: int, spare_asset_available: bool, forced: bool = False) -> int:
    if (not spare_asset_available) or forced:
        return max(1, severity_num - 1)
    return severity_num
```

## 5. RAC 매트릭스 — 결정론적, AI 배제

```python
RAC_MATRIX = {
    ("A", 1): "High", ("A", 2): "High", ("A", 3): "Serious", ("A", 4): "Medium",
    ("B", 1): "High", ("B", 2): "Serious", ("B", 3): "Medium", ("B", 4): "Low",
    ("C", 1): "Serious", ("C", 2): "Serious", ("C", 3): "Medium", ("C", 4): "Low",
    ("D", 1): "Serious", ("D", 2): "Medium", ("D", 3): "Low", ("D", 4): "Low",
    ("E", 1): "Medium", ("E", 2): "Medium", ("E", 3): "Low", ("E", 4): "Low",
    ("F", 1): "Medium", ("F", 2): "Low", ("F", 3): "Low", ("F", 4): "Low",
}
RAC_ORDER = {"High": 1, "Serious": 2, "Medium": 3, "Low": 4}
```

## 6. AI 강화판 — 연속 L/S, 교차검증, 우선순위 점수

```python
CONFIDENCE_ANCHOR = 0.7
BASE_SCORE = {"Catastrophic": 0.90, "Critical": 0.60, "Marginal": 0.30, "Negligible": 0.10}
CONTINUOUS_S_CAP = 0.95
BATTERY_LOW_THRESHOLD = 30
LINK_QUALITY_LOW_THRESHOLD = 0.5
MARGIN_PENALTY_BATTERY = 0.10
MARGIN_PENALTY_SPARE_ASSET = 0.05
MARGIN_PENALTY_LINK = 0.05
KILL_CHAIN_URGENCY_BONUS = 0.1
AI_RELIABILITY_LOW_THRESHOLD = 2

def continuous_L(base_rate: float, confidence: float) -> float:
    scaled = base_rate * (confidence / CONFIDENCE_ANCHOR)
    cap = min(base_rate * 3, 0.95)
    return min(scaled, cap)

def margin_penalty(battery_pct, spare_asset_available: bool, link_quality) -> float:
    p = 0.0
    if battery_pct is not None and battery_pct < BATTERY_LOW_THRESHOLD:
        p += MARGIN_PENALTY_BATTERY
    if not spare_asset_available:
        p += MARGIN_PENALTY_SPARE_ASSET
    if link_quality is not None and link_quality < LINK_QUALITY_LOW_THRESHOLD:
        p += MARGIN_PENALTY_LINK
    return p

def continuous_S(severity_num_final: int, battery_pct, spare_asset_available: bool, link_quality) -> float:
    base = BASE_SCORE[SEVERITY_LABEL_BY_NUM[severity_num_final]]
    penalty = margin_penalty(battery_pct, spare_asset_available, link_quality)
    return min(base + penalty, CONTINUOUS_S_CAP)

def continuous_s_to_num(s: float) -> int:
    if s >= 0.75: return 1
    if s >= 0.45: return 2
    if s >= 0.20: return 3
    return 4

def ai_reliability(rac_det: str, rac_ai: str) -> str:
    diff = abs(RAC_ORDER[rac_det] - RAC_ORDER[rac_ai])
    return "low" if diff >= AI_RELIABILITY_LOW_THRESHOLD else "normal"

def compound_urgency_score(cont_L: float, cont_S: float, kill_chain_stage: str) -> float:
    bonus = KILL_CHAIN_URGENCY_BONUS if kill_chain_stage == "후기" else 0.0
    return min(cont_L * cont_S + bonus, 0.95)
```

`continuous_L`의 posture 보정은 결정론적 L과 **같은 `steps`를 재사용**해서 등급을 다시 이동시킵니다(아래 7절 `compute_risk_for_candidate` 참고) — 그렇지 않으면 posture가 격상된 상황에서 결정론적 RAC만 올라가고 AI측은 그대로라 `ai_reliability`가 항상 "low"로 뜨는 계산버그가 생깁니다. 이 문제는 설계 도중 발견해서 바로잡았습니다.

## 7. 후보 하나에 대한 전체 위험평가

```python
def compute_risk_for_candidate(candidate: dict, mission_brief: dict) -> dict:
    threat_event = candidate["threat_event"]
    potential_outcome = candidate["potential_outcome"]
    confidence = candidate["confidence"]
    kill_chain_stage = candidate["kill_chain_stage"]
    match_count = candidate["match_count"]
    forced_override = candidate.get("forced_severity_override", False)

    mission_context = mission_brief["mission_context"]
    posture = mission_brief["posture"]
    spare_asset_available = mission_brief.get("spare_asset_available", True)
    battery_pct = mission_brief.get("battery_pct")
    link_quality = mission_brief.get("link_quality")

    base_rate = get_base_rate(threat_event, mission_context)
    l_class_base = l_value_to_class(base_rate)
    steps = posture_shift_steps(posture, threat_event)
    l_class_final = shift_class(l_class_base, steps)

    severity_label = OUTCOME_TO_SEVERITY[potential_outcome]
    severity_num = SEVERITY_ORDER[severity_label]
    severity_num_final = apply_severity_override(severity_num, spare_asset_available, forced_override)

    rac = RAC_MATRIX[(l_class_final, severity_num_final)]

    cont_L = continuous_L(base_rate, confidence)
    l_class_ai_raw = l_value_to_class(cont_L)
    l_class_ai_final = shift_class(l_class_ai_raw, steps)   # 결정론과 같은 steps

    cont_S = continuous_S(severity_num_final, battery_pct, spare_asset_available, link_quality)
    severity_num_ai = continuous_s_to_num(cont_S)

    rac_ai_equivalent = RAC_MATRIX[(l_class_ai_final, severity_num_ai)]
    reliability = ai_reliability(rac, rac_ai_equivalent)

    urgency = compound_urgency_score(cont_L, cont_S, kill_chain_stage)

    return {
        "threat_event": threat_event, "rac": rac,
        "l_class_final": l_class_final, "severity_label_final": SEVERITY_LABEL_BY_NUM[severity_num_final],
        "kill_chain_stage": kill_chain_stage, "match_count": match_count,
        "compound_risk_assessment": {
            "continuous_L": round(cont_L, 4), "continuous_S": round(cont_S, 4),
            "rac_ai_equivalent": rac_ai_equivalent, "ai_reliability": reliability,
        },
        "compound_urgency_score": round(urgency, 4),
        "_severity_num_final": severity_num_final,
    }
```

## 8. 전체 파이프라인 — candidates 전부 평가 + 정렬

```python
def run_risk_assessment(threat_modeling_result: dict, mission_brief: dict) -> dict:
    candidates_04 = threat_modeling_result.get("candidates", [])

    if not candidates_04:
        # issue #24 Lead 결정(2026-07-04): 위협 후보가 없으면 ambient_rac는 항상 "Low".
        # exposure≥0.7→Medium 승격 규칙은 폐기. background_exposure_score는 참고 지표로만 유지.
        return {"candidates": [], "ambient_rac": "Low"}

    results = [compute_risk_for_candidate(c, mission_brief) for c in candidates_04]
    results.sort(key=lambda r: (-r["compound_urgency_score"], r["_severity_num_final"], -r["match_count"]))
    for i, r in enumerate(results):
        r["priority_rank"] = i + 1
        r.pop("_severity_num_final", None)

    return {"candidates": results, "ambient_rac": None}
```

## 11. 손계산 검증 — 시나리오 3: T3+T4 동시발생

candidates 배열 처리·정렬 로직을 검증하기 위한 합성 시나리오입니다. 04 출력에 T3(시나리오1과 동일)와 T4(LOITER_ROI 국면배수 적용) 두 후보가 함께 들어옵니다.
